23struct CaseAttrs {
24    count: usize,
25    expr: Expr,
26}
27
28impl fmt::Debug for CaseAttrs {
29    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
30        formatter
31            .debug_struct("CaseAttrs")
32            .field("count", &self.count)
33            .finish_non_exhaustive()
34    }
35}
36
37impl CaseAttrs {
38    fn parse(attr: proc_macro2::TokenStream) -> syn::Result<Self> {
39        struct CaseAttrsSyntax {
40            count: LitInt,
41            _comma: Token![,],
42            expr: Expr,
43        }
44
45        impl Parse for CaseAttrsSyntax {
46            fn parse(input: ParseStream) -> syn::Result<Self> {
47                Ok(Self {
48                    count: input.parse()?,
49                    _comma: input.parse()?,
50                    expr: input.parse()?,
51                })
52            }
53        }
54
55        let syntax: CaseAttrsSyntax = syn::parse2(attr)?;
56        let count: usize = syntax.count.base10_parse()?;
57        if count == 0 {
58            let message = "number of test cases must be positive";
59            return Err(SynError::new(syntax.count.span(), message));
60        }
61        Ok(Self {
62            count,
63            expr: syntax.expr,
64        })
65    }
66}

68struct MapAttrs {
69    path: Option<Path>,
70}
71
72impl fmt::Debug for MapAttrs {
73    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
74        formatter
75            .debug_struct("MapAttrs")
76            .field("path", &self.path.as_ref().map(|_| "_"))
77            .finish()
78    }
79}
80
81impl MapAttrs {
82    fn map_arg(&self, arg: &Ident) -> proc_macro2::TokenStream {
83        if let Some(path) = &self.path {
84            quote!(#path(&#arg))
85        } else {
86            quote!(&#arg)
87        }
88    }
89}
90
91impl Parse for MapAttrs {
92    fn parse(input: ParseStream<'_>) -> syn::Result<Self> {
93        struct MapAttrsSyntax {
94            base: Ident,
95            path_expr: Option<(Token![=], Path)>,
96        }
97
98        impl Parse for MapAttrsSyntax {
99            fn parse(input: ParseStream<'_>) -> syn::Result<Self> {
100                Ok(Self {
101                    base: input.call(Ident::parse_any)?,
102                    path_expr: if input.peek(Token![=]) {
103                        Some((input.parse()?, input.parse()?))
104                    } else {
105                        None
106                    },
107                })
108            }
109        }
110
111        let syntax = MapAttrsSyntax::parse(input)?;
112        if syntax.base != "ref" {
113            let message = "unknown map transform; only `ref` is supported";
114            return Err(SynError::new(syntax.base.span(), message));
115        }
116
117        Ok(Self {
118            path: syntax.path_expr.map(|(_, path)| path),
119        })
120    }
121}

123struct FunctionWrapper {
124    #[cfg(feature = "nightly")]
125    nightly: NightlyData,
126    name: Ident,
127    attrs: CaseAttrs,
128    fn_attrs: Vec<Attribute>,
129    fn_sig: Signature,
130    arg_mappings: Vec<Option<MapAttrs>>,
131}
132
133impl fmt::Debug for FunctionWrapper {
134    fn fmt(&self, formatter: &mut fmt::Formatter<'_>) -> fmt::Result {
135        formatter
136            .debug_struct("FunctionWrapper")
137            .field("attrs", &self.attrs)
138            .field("name", &self.name)
139            .field("fn_attrs_len", &self.fn_attrs.len())
140            .finish_non_exhaustive()
141    }
142}

144impl FunctionWrapper {
145    const MAX_ARGS: usize = 7;
146
147    fn new(attrs: CaseAttrs, function: &mut ItemFn) -> syn::Result<Self> {
148        if function.sig.inputs.is_empty() {
149            let message = "tested function must have at least one arg";
150            return Err(SynError::new_spanned(&function.sig, message));
151        } else if function.sig.inputs.len() > Self::MAX_ARGS {
152            let message = format!(
153                "tested function must have no more than {} args",
154                Self::MAX_ARGS
155            );
156            return Err(SynError::new_spanned(&function.sig, message));
157        }
158
159        let generic_params = &function.sig.generics.params;
160        if !generic_params.is_empty() {
161            let message = "generic tested functions are not supported";
162            return Err(SynError::new_spanned(generic_params, message));
163        }
164
165        let mappings = function.sig.inputs.iter_mut().map(|arg| {
166            let attrs = match arg {
167                FnArg::Receiver(receiver) => &mut receiver.attrs,
168                FnArg::Typed(typed) => &mut typed.attrs,
169            };
170            let map_attr = attrs
171                .iter()
172                .enumerate()
173                .find(|(_, attr)| attr.path().is_ident("map"));
174            let Some((idx, map_attr)) = map_attr else {
175                return Ok(None);
176            };
177            let map_attr = map_attr.parse_args::<MapAttrs>()?;
178            attrs.remove(idx);
179            Ok(Some(map_attr))
180        });
181        let mappings: syn::Result<Vec<_>> = mappings.collect();
182        let mappings = mappings?;
183
184        let (retained_attrs, mut fn_attrs) = mem::take(&mut function.attrs)
185            .into_iter()
186            .partition(Self::should_be_retained);
187        function.attrs = retained_attrs;
188        let test_attr_position = fn_attrs
189            .iter()
190            .position(|attr| attr.path().is_ident("test"));
191        if cfg!(feature = "nightly") {
192            if let Some(position) = test_attr_position {
193                fn_attrs.remove(position);
194            }
195        } else if test_attr_position.is_none() && function.sig.asyncness.is_none() {
196            let test_attr = syn::parse_quote!(#[::core::prelude::v1::test]);
197            fn_attrs.insert(0, test_attr);
198        }
199
200        Ok(Self {
201            #[cfg(feature = "nightly")]
202            nightly: NightlyData::from_attrs(&mut fn_attrs)?,
203            name: function.sig.ident.clone(),
204            attrs,
205            fn_attrs,
206            fn_sig: function.sig.clone(),
207            arg_mappings: mappings,
208        })
209    }
210
211    // FIXME: this is extremely hacky. Ideally, we'd want to partition attrs by their location
212    //   before / after `#[test_casing]`, but this seems impossible on stable Rust (span locations
213    //   are unstable).
214    fn should_be_retained(attr: &Attribute) -> bool {
215        attr.path().is_ident("allow")
216            || attr.path().is_ident("warn")
217            || attr.path().is_ident("deny")
218            || attr.path().is_ident("forbid")
219    }
220
221    fn arg_names(&self) -> impl Iterator<Item = String> + '_ {
222        self.fn_sig
223            .inputs
224            .iter()
225            .enumerate()
226            .map(|(i, arg)| match arg {
227                FnArg::Receiver(_) => String::from("self"),
228                FnArg::Typed(PatType { pat, .. }) => {
229                    if let Pat::Ident(ident) = pat.as_ref() {
230                        ident.ident.to_string()
231                    } else {
232                        format!("(arg {i})")
233                    }
234                }
235            })
236    }

238    fn test_cases_iter(&self) -> impl ToTokens {
239        let cr = quote!(::test_casing);
240        let name = &self.name;
241        let cases_expr = &self.attrs.expr;
242        let (arg_idents, case_args) = self.case_binding();
243        let case_binding = Self::group_idents(&arg_idents);
244        let maybe_output_binding = match (&self.fn_sig.asyncness, &self.fn_sig.output) {
245            (None, ReturnType::Default) => None,
246            _ => Some(quote!(let _ = )),
247        };
248        // ^ Using `let _ = ` on the `()` return type triggers https://rust-lang.github.io/rust-clippy/master/index.html#/ignored_unit_patterns
249        // in Rust 1.73+.
250
251        quote! {
252            const _: () = {
253                #[allow(dead_code, clippy::no_effect_underscore_binding)]
254                fn __test_cases_iterator() {
255                    let #case_binding = #cr::_private::case(#cases_expr, 0);
256                    #maybe_output_binding #name(#case_args);
257                }
258            };
259        }
260    }
261
262    fn wrap(&self) -> impl ToTokens {
263        let name = &self.name;
264        let test_cases_iter = self.test_cases_iter();
265        let arg_names = self.arg_names();
266        let arg_count = self.fn_sig.inputs.len();
267        let arg_names = quote! {
268            const __ARG_NAMES: [&'static str; #arg_count] = [#(#arg_names,)*];
269        };
270
271        let index_width = (self.attrs.count - 1).to_string().len();
272        let cases = (0..self.attrs.count).map(|i| self.case(i, index_width));
273
274        // Ideally, we'd want to assert on the cases iterator length in compile time.
275        // Since this is impossible on stable Rust, we do the next best thing - generating a dedicated test for this.
276        let count = self.attrs.count;
277        let cases_expr = &self.attrs.expr;
278        let cr = quote!(::test_casing);
279        let case_count_assert = quote! {
280            #[test]
281            fn case_count_is_correct() {
282                #cr::_private::assert_case_count(#cases_expr, #count);
283            }
284        };
285
286        quote! {
287            // Access the iterator to ensure it works even if not building for tests.
288            #test_cases_iter
289
290            #[cfg(test)]
291            #[allow(clippy::no_effect_underscore_binding)]
292            // ^ We use `__ident`s to not alias user-defined idents accidentally. Unfortunately,
293            // this triggers this lint on Rust 1.76+.
294            mod #name {
295                use super::*;
296                #arg_names
297                #case_count_assert
298                #(#cases)*
299            }
300        }
301    }

303    #[cfg(feature = "nightly")]
304    fn declare_test_case(&self, index: usize, test_fn_name: &Ident) -> impl ToTokens {
305        let cr = quote!(::test_casing);
306        let cases_expr = &self.attrs.expr;
307        let test_case_name = format!("__TEST_CASE_{index}");
308        let test_case_name = Ident::new(&test_case_name, self.name.span());
309        let additional_args = self.nightly.macro_args();
310
311        let span_start = self.name.span().start();
312        let start_line = span_start.line;
313        let start_col = span_start.column;
314        let span_end = self.name.span().end();
315        let end_line = span_end.line;
316        let end_col = span_end.column;
317
318        quote! {
319            #[::core::prelude::v1::test_case]
320            static #test_case_name: #cr::nightly::LazyTestCase = #cr::declare_test_case!(
321                base_name: ::core::module_path!(),
322                source_file: ::core::file!(),
323                start_line: #start_line,
324                start_col: #start_col,
325                end_line: #end_line,
326                end_col: #end_col,
327                arg_names: __ARG_NAMES,
328                cases: #cases_expr,
329                index: #index,
330                #additional_args
331                testfn: #test_fn_name
332            );
333        }
334    }

336    fn case(&self, index: usize, index_width: usize) -> impl ToTokens {
337        let case_name = format!("case_{index:0>index_width$}");
338        let case_name = Ident::new(&case_name, self.name.span());
339
340        #[cfg(feature = "nightly")]
341        {
342            let case_fn = self.case_fn(index, &case_name);
343            let test_fn_name = format!("__TEST_FN_{index}");
344            let test_fn_name = Ident::new(&test_fn_name, self.name.span());
345            let ret = &self.fn_sig.output;
346            let case_decl = self.declare_test_case(index, &test_fn_name);
347
348            quote! {
349                #[allow(unnameable_test_items)]
350                // ^ This is a very roundabout way to effectively drop the `#[test]` attribute
351                // from the generated code. It should work for all kinds of test macros,
352                // such as `async_std::test` or `tokio::test`, without any additional work.
353                const #test_fn_name: fn() #ret = {
354                    #case_fn
355                    #case_name
356                };
357                #case_decl
358            }
359        }
360
361        #[cfg(not(feature = "nightly"))]
362        self.case_fn(index, &case_name)
363    }
364
365    fn case_fn(&self, index: usize, case_name: &Ident) -> proc_macro2::TokenStream {
366        let cr = quote!(::test_casing);
367        let name = &self.name;
368        let attrs = &self.fn_attrs;
369
370        let maybe_async = &self.fn_sig.asyncness;
371        let maybe_await = maybe_async.as_ref().map(|_| quote!(.await));
372        let ret = &self.fn_sig.output;
373        let maybe_semicolon = match ret {
374            ReturnType::Default => Some(quote!(;)),
375            ReturnType::Type { .. } => None,
376        };
377        let cases_expr = &self.attrs.expr;
378        let (arg_idents, case_args) = self.case_binding();
379        let case_binding = Self::group_idents(&arg_idents);
380
381        let case_assignment = if cfg!(feature = "nightly") {
382            quote! {
383                let #case_binding = #cr::_private::case(#cases_expr, #index);
384            }
385        } else {
386            quote! {
387                let __case = #cr::_private::case(#cases_expr, #index);
388                let #case_binding = __case;
389            }
390        };
391
392        let args = self
393            .arg_names()
394            .zip(arg_idents)
395            .map(|(name, arg)| quote!(#name = #arg));
396
397        quote! {
398            #(#attrs)*
399            #maybe_async fn #case_name() #ret {
400                #case_assignment
401                let __guard = #cr::__describe_test_case!(#name, #index, #(#args,)*);
402                #name(#case_args) #maybe_await #maybe_semicolon
403            }
404        }
405    }
406
407    /// Returns the binding of args supplied to the test case and potentially mapped args
408    /// to provide to the test function.
409    fn case_binding(&self) -> (Vec<Ident>, impl ToTokens) {
410        if self.fn_sig.inputs.len() == 1 {
411            let arg = self.fn_sig.inputs.first().unwrap();
412            let arg = Ident::new("__case_arg", arg.span());
413            let mapped_arg = self.arg_mappings[0]
414                .as_ref()
415                .map_or_else(|| quote!(#arg), |mapping| mapping.map_arg(&arg));
416            (vec![arg], mapped_arg)
417        } else {
418            let args = self.fn_sig.inputs.iter().enumerate();
419            let args = args.map(|(idx, arg)| Ident::new(&format!("__case_arg{idx}"), arg.span()));
420            let case_binding = args.clone().collect();
421
422            let args = args.zip(&self.arg_mappings).map(|(arg, mapping)| {
423                mapping
424                    .as_ref()
425                    .map_or_else(|| quote!(#arg), |mapping| mapping.map_arg(&arg))
426            });
427            let case_args = quote!(#(#args,)*);
428            (case_binding, case_args)
429        }
430    }
431
432    fn group_idents(arg_idents: &[Ident]) -> impl ToTokens {
433        if arg_idents.len() == 1 {
434            let arg = &arg_idents[0];
435            quote!(#arg)
436        } else {
437            let arg_idents = arg_idents.iter();
438            quote!((#(#arg_idents,)*))
439        }
440    }
441}

443pub(crate) fn impl_test_casing(
444    attr: TokenStream,
445    item: TokenStream,
446) -> syn::Result<proc_macro2::TokenStream> {
447    let attrs = CaseAttrs::parse(attr.into())?;
448    let item: Item = syn::parse(item)?;
449    match item {
450        Item::Fn(mut function) => {
451            let wrapper = FunctionWrapper::new(attrs, &mut function)?;
452            let wrapper = wrapper.wrap();
453            Ok(quote!(#function #wrapper))
454        }
455        item => {
456            let message = "Item is not supported; use `#[test_cases] on functions";
457            Err(SynError::new_spanned(&item, message))
458        }
459    }
